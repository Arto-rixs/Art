if:

    if - это условная конструкция, которая выполняет определенный блок кода только в том случае, если указанное условие истинно.
    if (условие) {
    // Блок кода, выполняемый, если условие истинно
}
Пример:
int x = 10;
if (x > 5) {
    cout << "x больше 5";
}

if и else являются ключевыми конструкциями условного оператора в языке программирования C++.

Конструкция if используется для выполнения определенного блока кода только в случае, если указанное условие истинно. Конструкция else предоставляет альтернативный блок кода, который выполняется, если условие в if ложно.

Вот пример использования конструкции if else:

int x = 10;
if (x > 5) {
    cout << "x больше 5" << endl;
} else {
    cout << "x меньше или равно 5" << endl;
}


for:

    for - это циклическая конструкция, которая выполняет определенный блок кода заданное количество раз.
    Формат:
    for (инициализация; условие; обновление) {
    // Блок кода, выполняемый в каждой итерации цикла
}
Пример:
for (int i = 0; i < 5; i++) {
    cout << i << endl;
}


while:

    while - это циклическая конструкция, которая выполняет определенный блок кода до тех пор, пока указанное условие истинно.
    Формат:

while (условие) {
    // Блок кода, выполняемый в каждой итерации цикла
}
Пример:
int i = 0;
while (i < 5) {
    cout << i << endl;
    i++;
}
